\hypertarget{mappable-actions}{%
\subsection{Mappable actions}\label{mappable-actions}}

\hypertarget{copypaste}{%
\subsubsection{Copy/Paste}\label{copypaste}}

\begin{itemize}
\tightlist
\item
  \textbf{copy\_and\_clear\_or\_interrupt}: Copy the selected text from
  the active window to the clipboard and clear selection, if no
  selection, send Ctrl-C
\item
  \textbf{copy\_or\_interrupt}: Copy the selected text from the active
  window to the clipboard, if no selection, send Ctrl-C
\item
  \textbf{copy\_to\_clipboard}: Copy the selected text from the active
  window to the clipboard. Default shortcut: ctrl+shift+c
\item
  \textbf{pass\_selection\_to\_program}: Pass selected text from active
  window to specified program. Default shortcut: ctrl+shift+o
\item
  \textbf{paste}: Paste the specified text into the current window
\item
  \textbf{show\_scrollback}: Show scrollback in a pager like less.
  Default shortcut: ctrl+shift+h
\item
  \textbf{copy\_to\_buffer}: Copy the selection from the active window
  to the specified buffer.
\item
  \textbf{paste\_from\_buffer}: Paste from the specified buffer to the
  active window.
\item
  \textbf{paste\_from\_clipboard}: Paste from the clipboard to the
  active window. Default shortcut: ctrl+shift+v
\item
  \textbf{paste\_from\_selection}: Paste from the clipboard to the
  active window. Default shortcut: ctrl+shift+s
\end{itemize}

\hypertarget{mouse-actions}{%
\subsubsection{Mouse actions}\label{mouse-actions}}

\begin{itemize}
\tightlist
\item
  \textbf{mouse\_click\_url}: Click the URL under the mouse
\item
  \textbf{mouse\_click\_url\_or\_select}: Click the URL under the mouse
  only if the screen has no selection
\item
  \textbf{mouse\_selection}: Manipulate the selection based on the
  current mouse position
\item
  \textbf{paste\_selection}: Paste the current primary selection
\item
  \textbf{paste\_selection\_or\_clipboard}: Paste the current primary
  selection or the clipboard if no selection is present
\end{itemize}

\hypertarget{scrolling-1}{%
\subsubsection{Scrolling}\label{scrolling-1}}

\begin{itemize}
\tightlist
\item
  \textbf{scroll\_end}: Scroll to the bottom of the scrollback buffer.
  Default shortcut: ctrl+shift+end
\item
  \textbf{scroll\_home}: Scroll to the top of the scrollback buffer.
  Default shortcut: ctrl+shift+home
\item
  \textbf{scroll\_line\_down}: Scroll down by one line. Default
  shortcut: ctrl+shift+down
\item
  \textbf{scroll\_line\_up}: Scroll up by one line. Default shortcut:
  ctrl+shift+up
\item
  \textbf{scroll\_page\_down}: Scroll down by one page. Default
  shortcut: ctrl+shift+page\_down
\item
  \textbf{scroll\_page\_up}: Scroll up by one page. Default shortcut:
  ctrl+shift+page\_up
\end{itemize}

\hypertarget{windows-management-1}{%
\subsubsection{Windows management}\label{windows-management-1}}

\begin{itemize}
\tightlist
\item
  \textbf{close\_other\_windows\_in\_tab}: Close all windows in the tab
  other than the currently active window.
\item
  \textbf{close\_window}: Close the currently active window. Default
  shortcut: ctrl+shift+w
\item
  \textbf{move\_window}: Move the window in the specified direction.
  Default shortcut: ctrl+shift+b
\item
  \textbf{move\_window\_forward}: Move active window forward (swap it
  with the next window). Default shortcut: ctrl+shift+f
\item
  \textbf{move\_window\_to\_top}: Move active window to the top (make it
  the first window). Default shortcut: ctrl+shift+`
\item
  \textbf{neighboring\_window}: Focus the neighboring window in the
  current tab. Default shortcut: ctrl+shift+{]}
\item
  \textbf{nth\_window}: Focus the nth window if positive or the
  previously active windows if negative. Default shortcut:
  ctrl+shift+{[}
\item
  \textbf{reset\_window\_sizes}: Reset window sizes undoing any dynamic
  resizing of windows.
\item
  \textbf{resize\_window}: Resize the active window by the specified
  amount.
\item
  \textbf{change\_font\_size}: Change the font size for the current or
  all OS Windows. Default shortcuts using this action: ctrl+shift+minus,
  ctrl+shift+equal, ctrl+shift+backspace
\item
  \textbf{close\_os\_window}: Close the currently active OS Window.
  Default shortcut: shift+cmd+w
\item
  \textbf{detach\_window}: Detach a window, moving it to another tab or
  OS Window.
\item
  \textbf{new\_os\_window\_with\_cwd}: New OS Window with the same
  working directory as the currently active window.
\item
  \textbf{new\_window}: Create a new window. Default shortcut:
  ctrl+shift+enter
\item
  \textbf{new\_window\_with\_cwd}: Create a new window with working
  directory same as that of the active window.
\item
  \textbf{quit}: Quit, closing all windows
\item
  \textbf{set\_background\_opacity}: Set the background opacity for the
  active OS Window. Default shortcuts using this action:
  ctrl+shift+a\textgreater l, ctrl+shift+a\textgreater1,
  ctrl+shift+a\textgreater m, ctrl+shift+a\textgreater d
\item
  \textbf{start\_resizing\_window}: Resize the active window
  interactively. Default shortcut: ctrl+shift+r
\item
  \textbf{toggle\_fullscreen}: Toggle the fullscreen status of the
  active OS Window. Default shortcut: ctrl+shift+f11
\item
  \textbf{toggle\_maximized}: Toggle the maximized status of the active
  OS Window. Default shortcut: ctrl+shift+f10
\end{itemize}

\hypertarget{tab-management}{%
\subsubsection{Tab management}\label{tab-management}}

\begin{itemize}
\tightlist
\item
  \textbf{detach\_tab}: Detach a tab, moving it to another OS Window
\item
  \textbf{goto\_tab}: Go to the specified tab, by number, starting with
  1. Zero and negative numbers go to previously active tabs
\item
  \textbf{move\_tab\_backward}: Move the active tab backward. Default
  shortcut: ctrl+shift+,
\item
  \textbf{move\_tab\_forward}: Move the active tab forward. Default
  shortcut: ctrl+shift+.
\item
  \textbf{new\_tab}: Create a new tab. Default shortcut: ctrl+shift+t
\item
  \textbf{new\_tab\_with\_cwd}: Create a new tab with working directory
  for the window in it set to the same as the active window.
\item
  \textbf{next\_tab}: Make the next tab active. Default shortcut:
  ctrl+shift+right
\item
  \textbf{previous\_tab}: Make the previous tab active. Default
  shortcut: ctrl+shift+left
\item
  \textbf{select\_tab}: Interactively select a tab to switch to
\item
  \textbf{set\_tab\_title}: Change the title of the active tab. Default
  shortcut: ctrl+shift+alt+t
\end{itemize}

\hypertarget{miscellaneous}{%
\subsubsection{Miscellaneous}\label{miscellaneous}}

\begin{itemize}
\tightlist
\item
  \textbf{send\_text}: Send the specified text to the active window.
  Default shortcut: ctrl+shift+alt+h
\item
  \textbf{signal\_child}: Send the specified SIGNAL to the foreground
  process in the active window.
\item
  \textbf{clear\_terminal}: Clear the terminal. Default shortcut:
  ctrl+shift+delete
\item
  \textbf{combine}: Combine multiple actions and map to a single
  keypress. The syntax is: map key combine action1 action2 action3
  \ldots{}
\item
  \textbf{disable\_ligatures\_in}: Turn on/off ligatures in the
  specified window.
\item
  \textbf{discard\_event}: Discard this event completely ignoring it.
\item
  \textbf{edit\_config\_file}: Edit the kitty.conf config file in your
  favorite text editor. Default shortcut: ctrl+shift+f2
\item
  \textbf{input\_unicode\_character}: Input an arbitrary unicode
  character.
\item
  \textbf{kitten}: Run the specified kitten. Default shortcuts using
  this action: ctrl+shift+p\textgreater n, ctrl+shift+u,
  ctrl+shift+p\textgreater h, ctrl+shift+p\textgreater l,
  ctrl+shift+p\textgreater f, ctrl+shift+p\textgreater w,
  ctrl+shift+p\textgreater y, ctrl+shift+p\textgreater shift+f
\item
  \textbf{kitty\_shell}: Run the kitty shell to control kitty with
  commands. Default shortcut: ctrl+shift+escape
\item
  \textbf{launch}: Launch the specified program in a new window/tab/etc.
\item
  \textbf{load\_config\_file}: Reload the config file. Default shortcut:
  ctrl+shift+f5
\item
  \textbf{open\_url\_with\_hints}: Click a URL using the keyboard.
  Default shortcut: ctrl+shift+e
\item
  \textbf{remote\_control}: Run a remote control command.
\item
  \textbf{set\_colors}: Change colors in the specified windows.
\item
  \textbf{show\_kitty\_env\_vars}: Show the environment variables that
  the kitty process sees.
\end{itemize}
